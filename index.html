${p.meno}
${p.hodnost} • #${p.badge} (${p.callsign})
PROFIL >
`; container.appendChild(card); }); } // Zobrazenie detailu (Discord Nick vidí iba Admin) function openProfile(p) { const modal = document.getElementById('profile-modal'); document.getElementById('modal-data').innerHTML = `

` : ''}
EVOC${p.evoc}
Zbrane${p.zbrane}
SEB${p.seb}
Medical/ASD${p.medical}
DOB${p.dob}
Badge#${p.badge}
`; modal.style.display = 'flex'; } // Pridanie člena (doplnené o nové polia) async function submitMember(action) { const d = { action, meno: document.getElementById('new-meno').value, badge: document.getElementById('new-badge').value, hodnost: document.getElementById('new-hodnost').value, discord: document.getElementById('new-discord').value, callsign: document.getElementById('new-callsign').value, dob: document.getElementById('new-dob').value, tel: document.getElementById('new-tel').value, foto: document.getElementById('new-foto').value }; // ... zvyšok fetch funkcie ... }
